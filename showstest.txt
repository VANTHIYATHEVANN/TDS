it("Should render image inside Avatar tag", () => {
        const { getByAltText } = render(<Shows history={testHistory} location={testLocation} isAdmin={true}/>);

        const movie1Avatar = getByAltText("Movie 1");
        const movie2Avatar = getByAltText("Movie 2");

        expect(movie1Avatar).toBeInTheDocument();
        expect(movie2Avatar).toBeInTheDocument();
    });

it("Should render image inside Dialog when opened", () => {
    const { getByAltText, getByTestId } = render(<Shows history={testHistory} location={testLocation} isAdmin={true}/>);

    // Initially, dialog should not be open
    expect(queryByTestId("avatar-dialog")).toBeNull();

    fireEvent.click(getByAltText("Movie 1")); // Open the dialog

    const dialog = getByTestId("avatar-dialog");
    expect(dialog).toBeInTheDocument();

    const avatarImage = getByAltText("Movie 1"); // Find the image inside dialog
    expect(avatarImage).toBeInTheDocument();
});
